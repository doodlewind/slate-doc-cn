A simulator to help writing tests for Slate editors and plugins.

Example

import Simulator from 'slate-simulator'

const state = ...
const plugins = [ ... ]
const simulator = new Simulator({ state, plugins })

simulator
  .focus()
  .beforeInput({ data: 'H' })
  .beforeInput({ data: 'e' })
  .beforeInput({ data: 'l' })
  .beforeInput({ data: 'l' })
  .beforeInput({ data: 'o' })
  .beforeInput({ data: '!' })
  .keyDown({}, { key: 'enter' })

const nextState = simulator.state
